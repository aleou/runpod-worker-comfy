#!/usr/bin/env bash
set -euo pipefail

BASE="/comfyui/models"

download() {
  local dest="$1" url="$2"
  mkdir -p "$BASE/$dest"
  local fname="$(basename "${url%%\?*}")"
  echo "→ $dest/$fname"
  curl -L --retry 5 --fail --progress-bar "$url" -o "$BASE/$dest/$fname"
}
